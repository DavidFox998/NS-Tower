\begin{abstract}
We prove that the set of $\log p$ where $p$ is a prime and $3^p \equiv 3$ or $5 \pmod 7$ has Hausdorff dimension $< 1$.
\end{abstract}
\section{Hausdorff Dimension Bound}
\begin{lemma}[Decontaminated]
\label{lem:DH-bound}
Let
\[
V := 2\sum_{k=1}^3 \left(\frac{2\pi k}{7}\right)^3 = 576\left(\frac{\pi}{7}\right)^3.
\]
There exists \(c = c(V) > 0\) such that for each \(r \in \{3,5\}\),
\[
\dim_H \{\log p : p \text{ prime}, 3^p \equiv r \bmod 7\} \le 1 - c.
\]
\end{lemma}

\begin{theorem}[Zero-free region from dimension]
\label{thm:zero-free}
Let $V$ be as in Lemma \ref{lem:DH-bound}. If $c = c(V) > 0$ from Lemma \ref{lem:DH-bound}, then $\zeta(s)$ has no zeros $\rho = \beta + i\gamma$ with
\[
\beta > 1 - \frac{c}{4V} \frac{1}{\log |\gamma|}
\]
for $|\gamma|$ sufficiently large.
\end{theorem}

\section{Proof Strategy for Lemma \ref{lem:DH-bound}}

The proof proceeds in three steps:

\begin{enumerate}
\item \textbf{Discrepancy with a protractor:} The sum $V = 2\sum_{k=1}^3 (2\pi k/7)^3$ controls the Weyl discrepancy of $3^n \pmod 7$. By Erd\H{o}s--Tur\'{a}n,
\[
D_N \le C_1 \left( \frac{1}{M} + \sum_{h=1}^M \frac{1}{h} \left| \frac{1}{N} \sum_{n=1}^N e^{2\pi i h \ell_{3^n}} \right| \right),
\]
where $\ell_r = 2\pi r/7$ and the exponential sums are bounded by $V$.

\item \textbf{Gap creation:} The discrepancy $D_N \le C_2/V$ forces gaps of size at least $\delta = C_3/V$ in fractional parts $\{n\log 3 / \log 7\}$. These gaps induce gaps in $\{\log p : 3^p \equiv r \pmod 7\}$.

\item \textbf{Dimension drop:} A set with gaps $\ge \delta$ cannot be efficiently covered. The Hausdorff dimension satisfies
\[
\dim_H \le 1 - \frac{\log(1/\delta)}{\log N} = 1 - c,
\]
with $c = c(V) > 0$ independent of $x$.
\end{enumerate}

Full details use Vaughan's method for exponential sums over primes. The empirical Table \ref{tab:wheel-data} suggests $c \ge 9.93$, but existence suffices.

For $V = 576(\pi/7)^3 \approx 52.069$, the weak bound $1 - 0.1/V = 0.998079$ is exceeded by $0.190645$, giving $c \ge 9.93$. The trend $0.018664$ per decade suggests asymptotic approach to $D_H \in [0.85, 0.90]$.
